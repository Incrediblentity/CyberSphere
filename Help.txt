import React, { useState, useEffect } from "react";
import { motion } from "framer-motion";
import { MapPin, Phone, Shield, AlertCircle, Navigation, Loader2 } from "lucide-react";
import { Card } from "@/components/ui/card";
import { Button } from "@/components/ui/button";
import { base44 } from "@/api/base44Client";

export default function Help() {
  const [location, setLocation] = useState(null);
  const [loading, setLoading] = useState(false);
  const [contacts, setContacts] = useState(null);
  const [error, setError] = useState(null);

  const getLocation = () => {
    setLoading(true);
    setError(null);

    if (!navigator.geolocation) {
      setError("Geolocation is not supported by your browser");
      setLoading(false);
      return;
    }

    navigator.geolocation.getCurrentPosition(
      async (position) => {
        const { latitude, longitude } = position.coords;
        setLocation({ latitude, longitude });

        try {
          const result = await base44.integrations.Core.InvokeLLM({
            prompt: `Find emergency cyber crime and police contact information for this location: ${latitude}, ${longitude}
            
            Provide:
            1. Local police station phone number
            2. Cyber crime cell/unit phone number
            3. National cyber crime helpline
            4. Emergency response number
            5. City or region name
            
            Format as structured data with clear contact information.`,
            add_context_from_internet: true,
            response_json_schema: {
              type: "object",
              properties: {
                city: { type: "string" },
                local_police: { type: "string" },
                cyber_crime_unit: { type: "string" },
                national_helpline: { type: "string" },
                emergency_number: { type: "string" }
              }
            }
          });

          setContacts(result);
        } catch (err) {
          setError("Failed to fetch contact information");
        }

        setLoading(false);
      },
      (err) => {
        setError("Unable to retrieve your location");
        setLoading(false);
      }
    );
  };

  useEffect(() => {
    getLocation();
  }, []);

  return (
    <div className="max-w-5xl mx-auto px-6 py-12">
      {/* Hero Section */}
      <motion.div
        initial={{ opacity: 0, y: -20 }}
        animate={{ opacity: 1, y: 0 }}
        className="text-center mb-16"
      >
        <div className="inline-flex items-center gap-2 bg-red-500/10 border border-red-500/30 rounded-full px-6 py-2 mb-6">
          <AlertCircle className="w-4 h-4 text-red-400" />
          <span className="text-red-400 text-sm font-medium">Emergency Assistance</span>
        </div>
        
        <h1 className="text-5xl md:text-6xl font-bold mb-6 bg-gradient-to-r from-red-400 via-orange-400 to-yellow-400 bg-clip-text text-transparent">
          Get Immediate Help
        </h1>
        
        <p className="text-xl opacity-80 max-w-3xl mx-auto">
          If you've encountered a cyber threat, time is critical. Access local authorities 
          and cyber crime units instantly.
        </p>
      </motion.div>

      {/* Location Card */}
      <motion.div
        initial={{ opacity: 0, scale: 0.95 }}
        animate={{ opacity: 1, scale: 1 }}
        transition={{ delay: 0.1 }}
        className="mb-8"
      >
        <Card className="backdrop-blur-xl bg-white/5 border-white/10 p-8 rounded-2xl shadow-2xl">
          <div className="flex items-center justify-between mb-6">
            <div className="flex items-center gap-3">
              <div className="w-12 h-12 rounded-xl bg-gradient-to-br from-blue-400 to-blue-600 flex items-center justify-center shadow-lg">
                <MapPin className="w-6 h-6 text-white" />
              </div>
              <div>
                <h3 className="text-xl font-bold opacity-90">Your Location</h3>
                <p className="opacity-60 text-sm">
                  {location ? ${location.latitude.toFixed(4)}, ${location.longitude.toFixed(4)} : "Detecting..."}
                </p>
              </div>
            </div>
            <Button
              onClick={getLocation}
              disabled={loading}
              className="bg-gradient-to-r from-blue-500 to-blue-600 hover:from-blue-600 hover:to-blue-700 text-white"
            >
              {loading ? (
                <>
                  <Loader2 className="w-4 h-4 mr-2 animate-spin" />
                  Locating...
                </>
              ) : (
                <>
                  <Navigation className="w-4 h-4 mr-2" />
                  Refresh Location
                </>
              )}
            </Button>
          </div>

          {error && (
            <div className="bg-red-500/10 border border-red-500/30 rounded-lg p-4 text-red-300">
              {error}
            </div>
          )}
        </Card>
      </motion.div>

      {/* Emergency Contacts */}
      {contacts && (
        <motion.div
          initial={{ opacity: 0, y: 20 }}
          animate={{ opacity: 1, y: 0 }}
          transition={{ delay: 0.2 }}
          className="space-y-6"
        >
          <h2 className="text-3xl font-bold mb-6 flex items-center gap-3 opacity-90">
            <Phone className="w-8 h-8 text-teal-400" />
            Emergency Contacts - {contacts.city}
          </h2>

          <div className="grid md:grid-cols-2 gap-6">
            <Card className="backdrop-blur-xl bg-gradient-to-br from-teal-500/10 to-blue-500/10 border-teal-500/30 p-6 rounded-xl shadow-2xl">
              <div className="flex items-center gap-3 mb-4">
                <Shield className="w-8 h-8 text-teal-400" />
                <h4 className="text-xl font-bold opacity-90">Local Police Station</h4>
              </div>
              <a
                href={tel:${contacts.local_police}}
                className="text-3xl font-bold text-teal-400 hover:text-teal-300 transition-colors"
              >
                {contacts.local_police}
              </a>
            </Card>

            <Card className="backdrop-blur-xl bg-gradient-to-br from-purple-500/10 to-pink-500/10 border-purple-500/30 p-6 rounded-xl shadow-2xl">
              <div className="flex items-center gap-3 mb-4">
                <AlertCircle className="w-8 h-8 text-purple-400" />
                <h4 className="text-xl font-bold opacity-90">Cyber Crime Unit</h4>
              </div>
              <a
                href={tel:${contacts.cyber_crime_unit}}
                className="text-3xl font-bold text-purple-400 hover:text-purple-300 transition-colors"
              >
                {contacts.cyber_crime_unit}
              </a>
            </Card>

            <Card className="backdrop-blur-xl bg-gradient-to-br from-orange-500/10 to-red-500/10 border-orange-500/30 p-6 rounded-xl shadow-2xl">
              <div className="flex items-center gap-3 mb-4">
                <Phone className="w-8 h-8 text-orange-400" />
                <h4 className="text-xl font-bold opacity-90">National Helpline</h4>
              </div>
              <a
                href={tel:${contacts.national_helpline}}
                className="text-3xl font-bold text-orange-400 hover:text-orange-300 transition-colors"
              >
                {contacts.national_helpline}
              </a>
            </Card>

            <Card className="backdrop-blur-xl bg-gradient-to-br from-red-500/10 to-pink-500/10 border-red-500/30 p-6 rounded-xl shadow-2xl">
              <div className="flex items-center gap-3 mb-4">
                <AlertCircle className="w-8 h-8 text-red-400" />
                <h4 className="text-xl font-bold opacity-90">Emergency Number</h4>
              </div>
              <a
                href={tel:${contacts.emergency_number}}
                className="text-3xl font-bold text-red-400 hover:text-red-300 transition-colors"
              >
                {contacts.emergency_number}
              </a>
            </Card>
          </div>
        </motion.div>
      )}

      {/* Safety Tips */}
      <motion.div
        initial={{ opacity: 0, y: 20 }}
        animate={{ opacity: 1, y: 0 }}
        transition={{ delay: 0.3 }}
        className="mt-12"
      >
        <Card className="backdrop-blur-xl bg-white/5 border-white/10 p-8 rounded-2xl shadow-2xl">
          <h3 className="text-2xl font-bold mb-6 opacity-90">What to Do if You're a Victim</h3>
          <div className="space-y-4 opacity-80">
            <div className="flex items-start gap-3">
              <div className="w-6 h-6 rounded-full bg-teal-500 flex items-center justify-center flex-shrink-0 mt-1">
                <span className="text-white text-sm font-bold">1</span>
              </div>
              <p>Document all evidence immediately - screenshots, emails, messages, transaction records</p>
            </div>
            <div className="flex items-start gap-3">
              <div className="w-6 h-6 rounded-full bg-teal-500 flex items-center justify-center flex-shrink-0 mt-1">
                <span className="text-white text-sm font-bold">2</span>
              </div>
              <p>Do not delete any communications or files related to the incident</p>
            </div>
            <div className="flex items-start gap-3">
              <div className="w-6 h-6 rounded-full bg-teal-500 flex items-center justify-center flex-shrink-0 mt-1">
                <span className="text-white text-sm font-bold">3</span>
              </div>
              <p>Change all passwords for affected accounts immediately</p>
            </div>
            <div className="flex items-start gap-3">
              <div className="w-6 h-6 rounded-full bg-teal-500 flex items-center justify-center flex-shrink-0 mt-1">
                <span className="text-white text-sm font-bold">4</span>
              </div>
              <p>Contact your bank if financial information was compromised</p>
            </div>
            <div className="flex items-start gap-3">
              <div className="w-6 h-6 rounded-full bg-teal-500 flex items-center justify-center flex-shrink-0 mt-1">
                <span className="text-white text-sm font-bold">5</span>
              </div>
              <p>File an official report with local authorities using the contacts above</p>
            </div>
          </div>
        </Card>
      </motion.div>
    </div>
  );
}
